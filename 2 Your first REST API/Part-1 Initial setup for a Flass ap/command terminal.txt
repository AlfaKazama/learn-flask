# Ini adalah langkah yang digunaknan untuk membuat virtual environment

# buka terminal, lalu jalankan perintah dibawah, Ini akan menjalankan python executable, dan memintanya untuk menggunakan modul venv, dan virtual environment itu disebut .venv
python -m venv .venv    

# kemudian kita harus memberi tahu visual studio code, virtual environment mana yang akan digunaknan, tekan tombol seperti dibawah, kemudian cari python: select interpreter. lalu pilih Ptyhon 3.13.5 ('.venv': venv)
Ctrl + Shift + p


# lalu jalankan
.venv\Scripts\activate

# kemudian close dan open kembali terminal, jalankan perintah dibawah, setelah sesia kita telah menginstall flask di folder ".venv" pada project
pip install flask

# buat app.py

# perhatikan app.py, setalah membuat app object, jalankan perintah, flask run akan mencari file bernama "app.py" dan didalam "app.py" ia akan mencari variable bernama "app". Jadi penting bahwa nama file dan nama variablenya sama
$env:FLASK_APP = "app" (optional)
flask run

# setelah dijalankan kita akan meliha aplikasi berjalan: Running on http://127.0.0.1:5000, dan itu berarti bahwa ketika kita membuat request apapun, flask app akan membuatnya di alamat itu.
# Ketika kita mengirim data (send data) ke alamat itu, maka data tersebut akan diforward (diteruskan) ke flask app yang jika kita perintahkan kita dapat melakukan sesuatu dengannya

# ctrl + c untuk stop flask dan deactive untuk keluar dari environment
